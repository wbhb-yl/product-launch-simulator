#부모 테이블관계자식 테이블FK 컬럼설명ON DELETE
1users1 : Nproductscreated_by한 사용자가 여러 상품 생성RESTRICT
2feature_categories1 : Nfeaturescategory_id카테고리 1개에 여러 기능RESTRICT
3productsN : Mfeaturesproduct_features상품은 여러 기능 포함, 기능은 여러 상품에 사용CASCADE
4products1 : Nproduct_workflowsproduct_id상품 1개의 워크플로우 단계별 이력CASCADE
5workflow_steps1 : Nproduct_workflowsstep_id단계 정의 1개에 여러 실행 이력RESTRICT
6users1 : Nproduct_workflowsperformed_by각 단계를 수행한 담당자SET NULL
7products1 : Nreview_checklistsproduct_id상품 1개에 체크리스트 항목들CASCADE
8users1 : Nreview_checklistschecked_by체크 수행자SET NULL
9products1 : Napprovalsproduct_id상품의 승인 이력 (반려 후 재승인 가능)CASCADE
10users1 : Napprovalsapprover_id승인자RESTRICT
11products1 : Nproduct_featuresproduct_idN:M 해소 테이블 (제품→)CASCADE
12features1 : Nproduct_featuresfeature_idN:M 해소 테이블 (기능→)RESTRICT
users▼ 펼치기 
CREATE TABLE users (
  id          BIGINT       NOT NULL AUTO_INCREMENT,
  name        VARCHAR(60)  NOT NULL,
  email       VARCHAR(120) NOT NULL,
  role        ENUM('planner','reviewer','approver','admin') NOT NULL DEFAULT 'planner',
  is_active   BOOLEAN      NOT NULL DEFAULT TRUE,
  created_at  TIMESTAMP    NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  UNIQUE KEY  uq_users_email (email)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
feature_categories▼ 펼치기 
CREATE TABLE feature_categories (
  id          BIGINT      NOT NULL AUTO_INCREMENT,
  code        VARCHAR(30) NOT NULL,
  name        VARCHAR(60) NOT NULL,
  icon        VARCHAR(10),
  sort_order  INT         NOT NULL DEFAULT 0,
  PRIMARY KEY (id),
  UNIQUE KEY  uq_fc_code (code)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
features▼ 펼치기 
CREATE TABLE features (
  id           BIGINT       NOT NULL AUTO_INCREMENT,
  category_id  BIGINT       NOT NULL,
  code         VARCHAR(50)  NOT NULL,
  name         VARCHAR(80)  NOT NULL,
  icon         VARCHAR(10),
  description  VARCHAR(200),
  cost_amount  INT          NOT NULL DEFAULT 0,
  is_active    BOOLEAN      NOT NULL DEFAULT TRUE,
  created_at   TIMESTAMP    NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  UNIQUE KEY   uq_features_code (code),
  KEY          idx_features_active (is_active),
  CONSTRAINT   fk_features_category
    FOREIGN KEY (category_id) REFERENCES feature_categories(id)
    ON DELETE RESTRICT ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
products▼ 펼치기 
CREATE TABLE products (
  id               BIGINT      NOT NULL AUTO_INCREMENT,
  created_by       BIGINT      NOT NULL,
  name             VARCHAR(100)NOT NULL,
  description      TEXT,
  base_price       INT         NOT NULL,
  final_price      INT         NOT NULL,
  target_segment  ENUM('all','new','premium','enterprise') NOT NULL DEFAULT 'all',
  contract_type   ENUM('monthly','12month','24month') NOT NULL DEFAULT 'monthly',
  status          ENUM('draft','review','approved','launched','rejected') NOT NULL DEFAULT 'draft',
  launched_at     TIMESTAMP   NULL,
  created_at      TIMESTAMP   NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at      TIMESTAMP   NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  KEY          idx_products_status (status),
  CONSTRAINT   fk_products_user
    FOREIGN KEY (created_by) REFERENCES users(id)
    ON DELETE RESTRICT ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
product_features▼ 펼치기 
CREATE TABLE product_features (
  id             BIGINT    NOT NULL AUTO_INCREMENT,
  product_id     BIGINT    NOT NULL,
  feature_id     BIGINT    NOT NULL,
  cost_snapshot  INT       NOT NULL,  -- 선택 당시 비용 고정
  added_at       TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  UNIQUE KEY  uq_pf (product_id, feature_id),
  KEY         idx_pf_product (product_id),
  CONSTRAINT  fk_pf_product
    FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
  CONSTRAINT  fk_pf_feature
    FOREIGN KEY (feature_id) REFERENCES features(id) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
workflow_steps + product_workflows + review_checklists + approvals▼ 펼치기 
CREATE TABLE workflow_steps (
  id          BIGINT     NOT NULL AUTO_INCREMENT,
  code        VARCHAR(30)NOT NULL,
  name        VARCHAR(60)NOT NULL,
  description TEXT,
  sort_order  INT        NOT NULL,
  PRIMARY KEY (id),
  UNIQUE KEY  uq_ws_code (code)
);

CREATE TABLE product_workflows (
  id           BIGINT    NOT NULL AUTO_INCREMENT,
  product_id   BIGINT    NOT NULL,
  step_id      BIGINT    NOT NULL,
  performed_by BIGINT    NULL,
  status       ENUM('pending','in_progress','completed','rejected') NOT NULL DEFAULT 'pending',
  note         TEXT,
  started_at   TIMESTAMP NULL,
  completed_at TIMESTAMP NULL,
  PRIMARY KEY  (id),
  KEY          idx_pw_product (product_id),
  CONSTRAINT   fk_pw_product  FOREIGN KEY (product_id)   REFERENCES products(id)        ON DELETE CASCADE,
  CONSTRAINT   fk_pw_step     FOREIGN KEY (step_id)       REFERENCES workflow_steps(id) ON DELETE RESTRICT,
  CONSTRAINT   fk_pw_user     FOREIGN KEY (performed_by)  REFERENCES users(id)         ON DELETE SET NULL
);

CREATE TABLE review_checklists (
  id            BIGINT    NOT NULL AUTO_INCREMENT,
  product_id    BIGINT    NOT NULL,
  checklist_key ENUM('legal','pricing','network','marketing') NOT NULL,
  is_checked    BOOLEAN   NOT NULL DEFAULT FALSE,
  checked_by    BIGINT    NULL,
  checked_at    TIMESTAMP NULL,
  PRIMARY KEY   (id),
  UNIQUE KEY    uq_rc (product_id, checklist_key),
  KEY           idx_rc_product (product_id),
  CONSTRAINT    fk_rc_product  FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
  CONSTRAINT    fk_rc_user     FOREIGN KEY (checked_by) REFERENCES users(id)    ON DELETE SET NULL
);

CREATE TABLE approvals (
  id           BIGINT    NOT NULL AUTO_INCREMENT,
  product_id   BIGINT    NOT NULL,
  approver_id  BIGINT    NOT NULL,
  decision     ENUM('approved','rejected') NOT NULL,
  comment      TEXT,
  decided_at   TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY  (id),
  KEY          idx_ap_product (product_id),
  CONSTRAINT   fk_ap_product  FOREIGN KEY (product_id)  REFERENCES products(id) ON DELETE CASCADE,
  CONSTRAINT   fk_ap_approver FOREIGN KEY (approver_id) REFERENCES users(id)    ON DELETE RESTRICT
);
