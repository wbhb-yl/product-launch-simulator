CREATE TABLE products (
  id               BIGINT      NOT NULL AUTO_INCREMENT,
  created_by       BIGINT      NOT NULL,
  name             VARCHAR(100)NOT NULL,
  description      TEXT,
  base_price       INT         NOT NULL,
  final_price      INT         NOT NULL,
  target_segment  ENUM('all','new','premium','enterprise') NOT NULL DEFAULT 'all',
  contract_type   ENUM('monthly','12month','24month') NOT NULL DEFAULT 'monthly',
  status          ENUM('draft','review','approved','launched','rejected') NOT NULL DEFAULT 'draft',
  launched_at     TIMESTAMP   NULL,
  created_at      TIMESTAMP   NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at      TIMESTAMP   NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY (id),
  KEY          idx_products_status (status),
  CONSTRAINT   fk_products_user
    FOREIGN KEY (created_by) REFERENCES users(id)
    ON DELETE RESTRICT ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

  id             BIGINT    NOT NULL AUTO_INCREMENT,

  CONSTRAINT  fk_pf_feature
    FOREIGN KEY (feature_id) REFERENCES features(id) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE workflow_steps (
  id          BIGINT     NOT NULL AUTO_INCREMENT,
  code        VARCHAR(30)NOT NULL,
  name        VARCHAR(60)NOT NULL,
  description TEXT,
  sort_order  INT        NOT NULL,
  PRIMARY KEY (id),
  UNIQUE KEY  uq_ws_code (code)
);

CREATE TABLE product_workflows (
  id           BIGINT    NOT NULL AUTO_INCREMENT,
  product_id   BIGINT    NOT NULL,
  step_id      BIGINT    NOT NULL,
  performed_by BIGINT    NULL,
  status       ENUM('pending','in_progress','completed','rejected') NOT NULL DEFAULT 'pending',
  note         TEXT,
  started_at   TIMESTAMP NULL,
  completed_at TIMESTAMP NULL,
  PRIMARY KEY  (id),
  KEY          idx_pw_product (product_id),
  CONSTRAINT   fk_pw_product  FOREIGN KEY (product_id)   REFERENCES products(id)        ON DELETE CASCADE,
  CONSTRAINT   fk_pw_step     FOREIGN KEY (step_id)       REFERENCES workflow_steps(id) ON DELETE RESTRICT,
  CONSTRAINT   fk_pw_user     FOREIGN KEY (performed_by)  REFERENCES users(id)         ON DELETE SET NULL
);

CREATE TABLE review_checklists (
  id            BIGINT    NOT NULL AUTO_INCREMENT,

subscriptions▼ 펼치기
CREATE TABLE subscriptions (
  id               BIGINT      NOT NULL AUTO_INCREMENT,
  customer_id      BIGINT      NOT NULL,
  plan_id          BIGINT      NOT NULL,
  contract_type_id BIGINT      NOT NULL,
  status           ENUM('ordering','verifying','activating','active',
                        'suspended','terminating','terminated') NOT NULL DEFAULT 'ordering',
  pre_reserved_msisdn VARCHAR(15) NULL,                              -- ordering 단계 자동 배정, msisdn_pool.reserved 연동
  msisdn           VARCHAR(15) NULL,                                  -- activating 완료 후 확정
  sim_no           VARCHAR(20) NULL,
  monthly_price    INT         NULL,
  activated_at     TIMESTAMP   NULL,
  suspended_at     TIMESTAMP   NULL,
  terminated_at    TIMESTAMP   NULL,
  created_at       TIMESTAMP   NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY  (id),
  UNIQUE KEY   uq_sub_msisdn (msisdn),
  UNIQUE KEY   uq_sub_sim    (sim_no),
  KEY          idx_sub_customer (customer_id),
  KEY          idx_sub_status   (status),
  CONSTRAINT   fk_sub_customer  FOREIGN KEY (customer_id)      REFERENCES customers(id)      ON DELETE RESTRICT,
  CONSTRAINT   fk_sub_plan      FOREIGN KEY (plan_id)          REFERENCES plans(id)          ON DELETE RESTRICT,
  CONSTRAINT   fk_sub_contract  FOREIGN KEY (contract_type_id) REFERENCES contract_types(id) ON DELETE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
subscription_addons + lifecycle_events + msisdn_pool + sim_inventory▼ 펼치기
CREATE TABLE subscription_addons (
  id               BIGINT    NOT NULL AUTO_INCREMENT,
  subscription_id  BIGINT    NOT NULL,
  addon_id         BIGINT    NOT NULL,
  price_snapshot   INT       NOT NULL,
  added_at         TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
  removed_at       TIMESTAMP NULL,
  PRIMARY KEY  (id),
  KEY          idx_sa_sub   (subscription_id),
  CONSTRAINT   fk_sa_sub   FOREIGN KEY (subscription_id) REFERENCES subscriptions(id) ON DELETE CASCADE,
  CONSTRAINT   fk_sa_addon FOREIGN KEY (addon_id)         REFERENCES addons(id)         ON DELETE RESTRICT
);

CREATE TABLE lifecycle_events (
  id               BIGINT       NOT NULL AUTO_INCREMENT,
  subscription_id  BIGINT       NOT NULL,
  event_type       ENUM('order_created','number_reserved','identity_verified','activated',
                        'plan_changed','suspended','resumed','terminated') NOT NULL,
  from_status      ENUM('ordering','verifying','activating','active',
                        'suspended','terminating','terminated') NULL,
  to_status        ENUM('ordering','verifying','activating','active',
                        'suspended','terminating','terminated') NOT NULL,
  reason           VARCHAR(200) NULL,
  performed_at     TIMESTAMP    NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY  (id),
  KEY          idx_le_sub   (subscription_id),
  CONSTRAINT   fk_le_sub   FOREIGN KEY (subscription_id) REFERENCES subscriptions(id) ON DELETE CASCADE
);

CREATE TABLE msisdn_pool (
  id               BIGINT      NOT NULL AUTO_INCREMENT,
  msisdn           VARCHAR(15) NOT NULL,
  status           ENUM('available','reserved','assigned','released') NOT NULL DEFAULT 'available',  -- available→reserved→assigned→released
  subscription_id  BIGINT      NULL,
  reserved_at      TIMESTAMP   NULL,                                  -- ordering 단계 번호 자동 생성 시점
  assigned_at      TIMESTAMP   NULL,                                  -- activating 완료 후 확정 시점
  released_at      TIMESTAMP   NULL,                                  -- 해지 후 반납 시점
  PRIMARY KEY  (id),
  UNIQUE KEY   uq_mp_msisdn (msisdn),
  KEY          idx_mp_status (status),
  CONSTRAINT   fk_mp_sub    FOREIGN KEY (subscription_id) REFERENCES subscriptions(id) ON DELETE SET NULL
);

CREATE TABLE sim_inventory (
  id               BIGINT      NOT NULL AUTO_INCREMENT,
  sim_no           VARCHAR(20) NOT NULL,
  status           ENUM('available','issued','deactivated') NOT NULL DEFAULT 'available',
  subscription_id  BIGINT      NULL,
  issued_at        TIMESTAMP   NULL,
  returned_at      TIMESTAMP   NULL,
  PRIMARY KEY  (id),
  UNIQUE KEY   uq_si_sim    (sim_no),
  KEY          idx_si_status (status),
  CONSTRAINT   fk_si_sub    FOREIGN KEY (subscription_id) REFERENCES subscriptions(id) ON DELETE SET NULL
);
